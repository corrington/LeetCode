62     /*
63     A slightly improved brute force method.
64     Using the beginIndex and endIndex, a "window" is defined.
65     The window is slid along the string and substrings within the window
66     and extracted and tested. When the window hits the end of the string,
67     the window is shrunk by one and the process begins again.
68     Because the window size is going from large to small, the first palindrome
69     found is returned. If no palindrome exists, the first letter of the string
70     is returned.
71      */
72     public static String longestPalindrome2(String s) {
73         if (s.length() == 1) return s;
74 
75         // Define a "window" and slide it left to right across the string.
76         // After each iteration, make the window a little smaller
77         for (int windowSize = s.length() - 1; windowSize > 0; --windowSize) {
78             for (int beginIndex = 0; beginIndex < s.length() - windowSize; ++beginIndex) {
79                 int endIndex = beginIndex + windowSize + 1;
80                 // extract a substring from within the window and test it
81                 String candidate = s.substring(beginIndex, endIndex);
82                 if (isPalindrome(candidate)) {
83                     // the first palindrome found is the largest

85                 }
86             }
87         } // for window

94     public static String longestPalindrome(String s) {
95         if (s == null || s.length() < 1) return "";
96         int start = 0, end = 0;
97         for (int i = 0; i < s.length(); i++) {
98             int len1 = expandAroundCenter(s, i, i);               // check for the odd-length palindromes
99             int len2 = expandAroundCenter(s, i, i + 1);     //  check for the even-length palindromes
100             int len = Math.max(len1, len2);
101             if (len > end - start) {
102                 start = i - (len - 1) / 2;
103                 end = i + len / 2;
104             }
105         }
106         return s.substring(start, end + 1);
107     }

109     private static int expandAroundCenter(String s, int left, int right) {
110         int L = left, R = right;
111         while (L >= 0 && R < s.length() && s.charAt(L) == s.charAt(R)) {
112             L--;
113             R++;
114         }
115         return R - L - 1;
116     }
